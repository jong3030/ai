def get_secret(name: str, default: str = "") -> str:
    try:
        return st.secrets.get(name, default)
    except Exception:
        return default

class Store:
    def __init__(self):
        self.supabase_url = get_secret("SUPABASE_URL", "")
        self.supabase_key = get_secret("SUPABASE_SERVICE_KEY", "")
        self.mode = "local"
        self.sb = None
        if self.supabase_url and self.supabase_key and create_client:
            try:
                self.sb = create_client(self.supabase_url, self.supabase_key)
                self.mode = "supabase"
            except Exception as e:
                st.warning(f"Supabase 연결 실패. 임시 로컬 모드로 실행합니다: {e}")
        if self.mode == "local":
            self.conn = sqlite3.connect(LOCAL_DB, check_same_thread=False)
            self.init_local()

    def init_local(self):
        cur = self.conn.cursor()
        cur.execute("""
        create table if not exists materials (
            id integer primary key autoincrement,
            school text, grade text, subject text, material_type text, material_name text,
            importance integer default 50, created_at text
        )""")
        cur.execute("""
        create table if not exists problems (
            id integer primary key autoincrement,
            material_id integer,
            problem_kind text default 'material',
            exam_name text,
            page integer,
            number text,
            unit text,
            type text,
            difficulty text,
            problem_text text,
            solution_pattern text,
            source_note text,
            image_b64 text,
            created_at text
        )""")
        cur.execute("""
        create table if not exists match_candidates (
            id integer primary key autoincrement,
            exam_problem_id integer,
            candidate_problem_id integer,
            similarity real,
            weighted_score real,
            ai_grade text,
            final_grade text,
            status text default '후보',
            use_in_report integer default 0,
            case_order integer default 99,
            reason text,
            created_at text
        )""")
        cur.execute("""
        create table if not exists report_settings (
            id integer primary key autoincrement,
            exam_name text unique,
            title text,
            overview text,
            expected_cut text,
            key_questions text,
            study_guide text,
            updated_at text
        )""")
        self.conn.commit()

    def insert(self, table: str, row: Dict[str, Any]) -> Dict[str, Any]:
        if self.mode == "supabase":
            res = self.sb.table(table).insert(row).execute()
            return res.data[0] if res.data else row
        row = row.copy()
        now = datetime.now().isoformat(timespec="seconds")
        if table in ["materials", "problems", "match_candidates"] and "created_at" not in row:
            row["created_at"] = now
        if table == "report_settings" and "updated_at" not in row:
            row["updated_at"] = now
        cols = ",".join(row.keys())
        qs = ",".join(["?"] * len(row))
        cur = self.conn.cursor()
        cur.execute(f"insert into {table} ({cols}) values ({qs})", list(row.values()))
        self.conn.commit()
        row["id"] = cur.lastrowid
        return row

    def select(self, table: str, where: Optional[Dict[str, Any]] = None) -> pd.DataFrame:
        if self.mode == "supabase":
            q = self.sb.table(table).select("*")
            if where:
                for k, v in where.items():
                    q = q.eq(k, v)
            res = q.execute()
            return pd.DataFrame(res.data or [])
        sql = f"select * from {table}"
        vals = []
        if where:
            sql += " where " + " and ".join([f"{k}=?" for k in where.keys()])
            vals = list(where.values())
        return pd.read_sql_query(sql, self.conn, params=vals)

    def delete_all(self, table: str):
        if self.mode == "supabase":
            # Supabase requires a filter; delete broad by id > 0
            self.sb.table(table).delete().gt("id", 0).execute()
        else:
            self.conn.execute(f"delete from {table}")
            self.conn.commit()

    def update_row(self, table: str, row_id: int, updates: Dict[str, Any]):
        if self.mode == "supabase":
            self.sb.table(table).update(updates).eq("id", int(row_id)).execute()
        else:
            cols = ", ".join([f"{k}=?" for k in updates.keys()])
            vals = list(updates.values()) + [int(row_id)]
            self.conn.execute(f"update {table} set {cols} where id=?", vals)
            self.conn.commit()

store = Store()

def img_to_b64(img: Image.Image, max_width: int = 1400) -> str:
    img = img.convert("RGB")
    if img.width > max_width:
        ratio = max_width / img.width
        img = img.resize((max_width, int(img.height * ratio)))
    buf = io.BytesIO()
    img.save(buf, format="JPEG", quality=85)
    return base64.b64encode(buf.getvalue()).decode("utf-8")


def b64_to_img(b64: str) -> Optional[Image.Image]:
    if not b64:
        return None
    try:
        return Image.open(io.BytesIO(base64.b64decode(b64))).convert("RGB")
    except Exception:
        return None


def uploaded_to_pages(uploaded_file) -> List[Image.Image]:
    name = uploaded_file.name.lower()
    data = uploaded_file.read()
    pages = []
    if name.endswith(".pdf"):
        doc = fitz.open(stream=data, filetype="pdf")
        for page in doc:
            pix = page.get_pixmap(matrix=fitz.Matrix(2, 2), alpha=False)
            img = Image.open(io.BytesIO(pix.tobytes("png"))).convert("RGB")
            pages.append(img)
    else:
        img = Image.open(io.BytesIO(data)).convert("RGB")
        pages.append(ImageOps.exif_transpose(img))
    return pages


def crop_grid(img: Image.Image, rows: int, cols: int, margin_x: int = 0, margin_y: int = 0, gap_x: int = 0, gap_y: int = 0) -> List[Image.Image]:
    w, h = img.size
    inner_w = max(1, w - 2 * margin_x)
    inner_h = max(1, h - 2 * margin_y)
    cell_w = inner_w / cols
    cell_h = inner_h / rows
    crops = []
    for r in range(rows):
        for c in range(cols):
            left = int(margin_x + c * cell_w + gap_x / 2)
            top = int(margin_y + r * cell_h + gap_y / 2)
            right = int(margin_x + (c + 1) * cell_w - gap_x / 2)
            bottom = int(margin_y + (r + 1) * cell_h - gap_y / 2)
            crops.append(img.crop((left, top, right, bottom)))
    return crops


def manual_crop(img: Image.Image, left_pct: float, top_pct: float, width_pct: float, height_pct: float) -> Image.Image:
    w, h = img.size
    l = int(w * left_pct / 100)
    t = int(h * top_pct / 100)
    r = int(min(w, l + w * width_pct / 100))
    b = int(min(h, t + h * height_pct / 100))
    return img.crop((l, t, r, b))


def normalize_text(s: Any) -> str:
    if s is None:
        return ""
    return re.sub(r"\s+", " ", str(s)).strip()


def compose_search_text(row: pd.Series) -> str:
    return " ".join([
        normalize_text(row.get("unit", "")),
        normalize_text(row.get("type", "")),
        normalize_text(row.get("difficulty", "")),
        normalize_text(row.get("problem_text", "")),
        normalize_text(row.get("solution_pattern", "")),
        normalize_text(row.get("source_note", "")),
    ])


def grade_from_score(score: float) -> str:
    if score >= 0.82:
        return "A 직접 적중"
    if score >= 0.65:
        return "B 구조 적중"
    if score >= 0.45:
        return "C 개념 적중"
    return "D 제외"

def run_matching(exam_name: str, top_k: int = 5):
    store.delete_all("match_candidates")
    exam_df = store.select("problems", {"problem_kind":"exam"})
    if exam_name:
        exam_df = exam_df[exam_df["exam_name"].fillna("") == exam_name]
    mat_df = store.select("problems", {"problem_kind":"material"})
    materials = store.select("materials")
    if exam_df.empty or mat_df.empty:
        return 0
    mat_df = mat_df.merge(materials[["id","material_type","material_name"]], left_on="material_id", right_on="id", how="left", suffixes=("", "_mat"))
    exam_texts = [compose_search_text(r) for _, r in exam_df.iterrows()]
    mat_texts = [compose_search_text(r) for _, r in mat_df.iterrows()]
    vectorizer = TfidfVectorizer(analyzer="char_wb", ngram_range=(2, 5), min_df=1)
    X = vectorizer.fit_transform(exam_texts + mat_texts)
    sim = cosine_similarity(X[:len(exam_texts)], X[len(exam_texts):])
    count = 0
    for i, (_, erow) in enumerate(exam_df.iterrows()):
        scores = sim[i]
        cand_indices = scores.argsort()[::-1][:top_k]
        for rank, idx in enumerate(cand_indices, start=1):
            crow = mat_df.iloc[idx]
            base = float(scores[idx])
            weight = material_priority(crow.get("material_type", "")) / 100
            weighted = base * 0.72 + weight * 0.28
            ai_grade = grade_from_score(weighted)
            reason = f"단원/유형/핵심구조 유사도 기반 후보. 자료 우선순위: {crow.get('material_type','')}."
            store.insert("match_candidates", {
                "exam_problem_id": int(erow["id"]),
                "candidate_problem_id": int(crow["id"]),
                "similarity": round(base, 4),
                "weighted_score": round(weighted, 4),
                "ai_grade": ai_grade,
                "final_grade": ai_grade,
                "status": "후보",
                "use_in_report": 1 if rank == 1 and ai_grade != "D 제외" else 0,
                "case_order": rank if rank == 1 else 99,
                "reason": reason,
            })
            count += 1
    return count


def auto_approve_jikbo_first():
    matches = store.select("match_candidates")
    if matches.empty:
        return 0
    probs = store.select("problems")
    mats = store.select("materials")
    p = probs.merge(mats[["id","material_type","material_name"]], left_on="material_id", right_on="id", how="left", suffixes=("", "_mat"))
    match_full = matches.merge(p[["id","number","material_type","material_name"]], left_on="candidate_problem_id", right_on="id", how="left", suffixes=("", "_cand"))
    updated = 0
    for exam_pid, group in match_full.groupby("exam_problem_id"):
        group = group.copy()
        group["priority"] = group["material_type"].map(material_priority).fillna(50)
        group = group.sort_values(["priority", "weighted_score"], ascending=[False, False])
        for order, (_, row) in enumerate(group.iterrows(), start=1):
            use = 1 if order == 1 and row["final_grade"] != "D 제외" else 0
            status = "대표 적중" if use else "보조/후보"
            store.update_row("match_candidates", int(row["id"]), {"use_in_report": use, "status": status, "case_order": order if use else 99})
            updated += 1
    return updated

        hit_grade = ""
        hit_source = ""
        if not md.empty:
            m = md.sort_values("case_order").iloc[0]

                c = c.iloc[0]
                mat = mats[mats["id"] == c["material_id"]]
                mat_name = mat.iloc[0]["material_name"] if not mat.empty else ""
                mat_type = mat.iloc[0]["material_type"] if not mat.empty else ""
                hit = f"{mat_name} {c.get('number','')}번"
                hit_source = mat_type
                hit_grade = m.get("final_grade", "")
        rows.append({
            "번호": e.get("number", ""),
            "단원": e.get("unit", ""),
            "유형": e.get("type", ""),
            "난이도": e.get("difficulty", ""),
            "시험문항 요약": e.get("problem_text", ""),
            "대표 적중자료": hit,
            "자료구분": hit_source,
            "적중등급": hit_grade,
        })
    return pd.DataFrame(rows)


def generate_pdf(exam_name: str) -> Path:
    ensure_report(exam_name)
    settings = store.select("report_settings", {"exam_name": exam_name}).iloc[0]
    analysis = build_analysis_df(exam_name)
    path = OUTPUT_DIR / f"{exam_name}_분석지.pdf"
    pdfmetrics.registerFont(UnicodeCIDFont('HYSMyeongJo-Medium'))
    doc = SimpleDocTemplate(str(path), pagesize=A4, rightMargin=14*mm, leftMargin=14*mm, topMargin=14*mm, bottomMargin=14*mm)
    styles = getSampleStyleSheet()
    for s in styles.byName.values():
        s.fontName = 'HYSMyeongJo-Medium'
    story = []
    story.append(Paragraph(settings.get("title") or f"{exam_name} 시험 분석", styles["Title"]))
    story.append(Spacer(1, 6*mm))
    for title, body in [("시험 총평", settings.get("overview","")), ("예상 등급컷", settings.get("expected_cut","")), ("주요 문항 소개", settings.get("key_questions","")), ("향후 공부 방향", settings.get("study_guide",""))]:
        story.append(Paragraph(f"<b>{title}</b>", styles["Heading2"]))
        story.append(Paragraph(str(body).replace("\n", "<br/>"), styles["BodyText"]))
        story.append(Spacer(1, 4*mm))
    if not analysis.empty:
        story.append(Paragraph("문항별 분석표", styles["Heading2"]))
        table_data = [analysis.columns.tolist()] + analysis.fillna("").astype(str).values.tolist()
        # Trim long text for PDF table
        table_data = [[cell[:80] + ("…" if len(cell)>80 else "") for cell in row] for row in table_data]
        tbl = Table(table_data, repeatRows=1, colWidths=[16*mm, 28*mm, 35*mm, 18*mm, 45*mm, 40*mm, 20*mm, 25*mm])
        tbl.setStyle(TableStyle([
            ('FONT', (0,0), (-1,-1), 'HYSMyeongJo-Medium', 7),
            ('BACKGROUND', (0,0), (-1,0), colors.HexColor('#111827')),
            ('TEXTCOLOR', (0,0), (-1,0), colors.white),
            ('GRID', (0,0), (-1,-1), 0.25, colors.grey),
            ('VALIGN', (0,0), (-1,-1), 'TOP'),
        ]))
        story.append(tbl)
        story.append(PageBreak())
        hits = analysis[analysis["대표 적중자료"].astype(str) != ""]
        if not hits.empty:
            story.append(Paragraph("대표 적중문항", styles["Heading2"]))
            for _, r in hits.iterrows():
                story.append(Paragraph(f"<b>{r['번호']}번</b> — {r['대표 적중자료']} ({r['적중등급']})", styles["Heading3"]))
                story.append(Paragraph(f"유형: {r['유형']} / 난이도: {r['난이도']}<br/>요약: {r['시험문항 요약']}", styles["BodyText"]))
                story.append(Spacer(1, 3*mm))
    doc.build(story)
    return path


def generate_excel(exam_name: str) -> Path:
    path = OUTPUT_DIR / f"{exam_name}_분석표.xlsx"
    analysis = build_analysis_df(exam_name)
    matches = store.select("match_candidates")
    with pd.ExcelWriter(path, engine="openpyxl") as writer:
        analysis.to_excel(writer, sheet_name="문항별 분석", index=False)
        matches.to_excel(writer, sheet_name="적중 후보", index=False)
        store.select("problems").to_excel(writer, sheet_name="전체 문항DB", index=False)
    return path


def generate_ppt(exam_name: str) -> Path:
    ensure_report(exam_name)
    settings = store.select("report_settings", {"exam_name": exam_name}).iloc[0]
    analysis = build_analysis_df(exam_name)
    path = OUTPUT_DIR / f"{exam_name}_분석PPT.pptx"
    prs = Presentation()
    prs.slide_width = Inches(13.333)
    prs.slide_height = Inches(7.5)
    def add_title_slide(title, subtitle):
        slide = prs.slides.add_slide(prs.slide_layouts[6])
        bg = slide.background.fill
        bg.solid(); bg.fore_color.rgb = __import__('pptx').dml.color.RGBColor(17,24,39)
        tx = slide.shapes.add_textbox(Inches(0.7), Inches(1.8), Inches(12), Inches(1))
        p = tx.text_frame.paragraphs[0]; p.text = title; p.font.size = Pt(36); p.font.bold=True; p.font.color.rgb=__import__('pptx').dml.color.RGBColor(255,255,255)
        sub = slide.shapes.add_textbox(Inches(0.75), Inches(3.1), Inches(11.5), Inches(1.4))
        p2 = sub.text_frame.paragraphs[0]; p2.text = subtitle; p2.font.size=Pt(18); p2.font.color.rgb=__import__('pptx').dml.color.RGBColor(229,231,235)
    def add_text_slide(title, body):
        slide = prs.slides.add_slide(prs.slide_layouts[6])
        tx = slide.shapes.add_textbox(Inches(0.6), Inches(0.4), Inches(12.2), Inches(0.6))
        p=tx.text_frame.paragraphs[0]; p.text=title; p.font.size=Pt(26); p.font.bold=True
        bodybox=slide.shapes.add_textbox(Inches(0.8), Inches(1.4), Inches(11.8), Inches(5.5))
        tf=bodybox.text_frame; tf.word_wrap=True
        for i, line in enumerate(str(body).split('\n')):
            para = tf.paragraphs[0] if i==0 else tf.add_paragraph()
            para.text = line; para.font.size=Pt(18)
    add_title_slide(settings.get("title") or exam_name, "시험 총평 · 예상 등급컷 · 주요문항 · 적중문항 분석")
    add_text_slide("시험 총평", settings.get("overview", ""))
    add_text_slide("예상 등급컷", settings.get("expected_cut", ""))
    if not analysis.empty:
        hits = analysis[analysis["대표 적중자료"].astype(str) != ""]
        add_text_slide("대표 적중 요약", "\n".join([f"{r['번호']}번 → {r['대표 적중자료']} ({r['적중등급']})" for _, r in hits.head(8).iterrows()]))
        for _, r in hits.head(6).iterrows():
            add_text_slide(f"적중사례 | {r['번호']}번", f"유형: {r['유형']}\n난이도: {r['난이도']}\n대표 적중자료: {r['대표 적중자료']}\n판정: {r['적중등급']}\n\n{r['시험문항 요약']}")
    prs.save(path)
    return path
